# ‚úÖ **FAMILY TREE PROLOG CODE (SWISH Compatible)**

Copy‚Äìpaste this directly into **SWISH Prolog**:

```prolog
/* --- FACTS --- */

male(john).
male(raj).
male(ram).
male(ron).

female(mary).
female(sita).
female(ria).
female(kavya).

parent(john, raj).      % john is parent of raj
parent(mary, raj).

parent(john, ria).
parent(mary, ria).

parent(raj, ron).
parent(sita, ron).

parent(ria, kavya).
parent(ram, kavya).

/* --- BASIC RELATIONS --- */

father(F, C) :- male(F), parent(F, C).
mother(M, C) :- female(M), parent(M, C).

child(C, P) :- parent(P, C).

son(S, P) :- male(S), parent(P, S).
daughter(D, P) :- female(D), parent(P, D).

/* --- SIBLINGS --- */
sibling(A, B) :-
    parent(P, A), parent(P, B),
    A \= B.

brother(B, X) :- male(B), sibling(B, X).
sister(S, X) :- female(S), sibling(S, X).

/* --- GRAND RELATIONS --- */
grandparent(GP, C) :-
    parent(P, C), parent(GP, P).

grandfather(GF, C) :- male(GF), grandparent(GF, C).
grandmother(GM, C) :- female(GM), grandparent(GM, C).

/* --- AUNT & UNCLE --- */
uncle(U, X) :-
    brother(U, P), parent(P, X).

aunt(A, X) :-
    sister(A, P), parent(P, X).

/* --- COUSINS --- */
cousin(A, B) :-
    parent(P1, A),
    parent(P2, B),
    sibling(P1, P2),
    A \= B.
```
